	(function() {
		'use strict';

		function parseSize(txt) {
			txt = txt.trim();
			var m = txt.match(/^([\d]+(?:[.,]\d+)?)\s*(kg|g)\b/i);
			if (!m) return null;
			var num = parseFloat(m[1].replace(',', '.'));
			return m[2].toLowerCase() === 'kg' ? num * 1000 : num;
		}
		function ageWeight(txt) {
			var t = txt.toLowerCase();
			if (t.indexOf('jaun') !== -1) return 1;
			if (t.indexOf('suaug') !== -1) return 2;
			if (t.indexOf('senjor') !== -1 || t.indexOf('senst') !== -1) return 3;
			return 99;
		}
		function pkgWeight(txt) { var g = parseSize(txt); return g === null ? 999999 : g; }
		// Kramtuko dydis (cm grupes) - fiksuota tvarka
		function cmWeight(txt) {
			var t = txt.toLowerCase();
			if (t.indexOf('iki 10') !== -1) return 1;
			if (t.indexOf('10-15') !== -1) return 2;
			if (t.indexOf('16-25') !== -1) return 3;
			if (t.indexOf('virs 25') !== -1 || t.indexOf('25 cm') !== -1) return 4;
			return 99;
		}
		// Kiekis (vnt grupes) - fiksuota tvarka
		function vntWeight(txt) {
			var t = txt.toLowerCase();
			if (t.indexOf('1 vnt') !== -1 && t.indexOf('11') === -1) return 1;
			if (t.indexOf('2-5') !== -1) return 2;
			if (t.indexOf('6-10') !== -1) return 3;
			if (t.indexOf('11-25') !== -1) return 4;
			if (t.indexOf('26+') !== -1 || t.indexOf('26 ') !== -1) return 5;
			return 99;
		}
		// Dydis (XS..XXL) - fiksuota tvarka
		function sizeWeight(txt) {
			var t = txt.trim().toUpperCase();
			var m = { 'XS':0, 'S':1, 'M':2, 'L':3, 'XL':4, 'XXL':5 };
			return m.hasOwnProperty(t) ? m[t] : 99;
		}
		// Pavadelio ilgis (metrai) - skaitine tvarka, Automatinis i pabaiga
		function meterWeight(txt) {
			var t = txt.toLowerCase().trim();
			if (t.indexOf('automat') !== -1) return 999;
			var m = t.match(/(\d+(?:[.,]\d+)?)\s*m/);
			if (m) return parseFloat(m[1].replace(',', '.'));
			return 99;
		}
		function liText(li) {
			var l = li.querySelector('.term-label, label, a, span');
			return (l ? l.textContent : li.textContent).trim();
		}
		function applyOrder(filterEl, weightFn) {
			var content = filterEl.querySelector('.filter-content');
			if (!content) return;
			var list = content.querySelector('ul');
			if (!list) return;
			var items = Array.prototype.slice.call(list.children).filter(function(el){ return el.tagName === 'LI'; });
			if (items.length < 2) return;
			var isLabel = (list.className.indexOf('filter-label') !== -1);
			if (!isLabel) { list.style.display = 'flex'; list.style.flexDirection = 'column'; }
			var weighted = items.map(function(li){ return { li: li, w: weightFn(liText(li)) }; });
			weighted.sort(function(a, b){
				if (a.w === b.w) return liText(a.li).localeCompare(liText(b.li));
				return a.w - b.w;
			});
			weighted.forEach(function(o, idx){ o.li.style.order = idx; });
		}
		function orderOne(filterEl) {
			var tax = (filterEl.getAttribute('data-taxonomy') || '').toLowerCase();
			if (tax.indexOf('pavadelio_ilgis') !== -1) applyOrder(filterEl, meterWeight);
			else if (tax.indexOf('pakuotes') !== -1) applyOrder(filterEl, pkgWeight);
			else if (tax.indexOf('amzius') !== -1) applyOrder(filterEl, ageWeight);
			else if (tax === 'pa_dydis') applyOrder(filterEl, sizeWeight);
			else if (tax.indexOf('ilgis') !== -1) applyOrder(filterEl, cmWeight);
			else if (tax.indexOf('kiekis') !== -1) applyOrder(filterEl, vntWeight);
		}
		function orderAll() {
			document.querySelectorAll('.yith-wcan-filter[data-taxonomy]').forEach(orderOne);
		}

		function isMobile() { return window.innerWidth <= 768; }
		function titleText(f) { var h = f.querySelector('.filter-title'); return h ? h.textContent.trim() : ''; }
		function isTreatPage() {
			// skanestu kategorijos puslapis (URL ar body klase)
			var u = (location.pathname || '').toLowerCase();
			var b = (document.body.className || '').toLowerCase();
			return u.indexOf('skanest') !== -1 || u.indexOf('kramt') !== -1 ||
			       b.indexOf('skanest') !== -1 || b.indexOf('kramt') !== -1;
		}
		function isVitaminPage() {
			// vitaminu/papildu kategorijos puslapis (URL ar body klase)
			var u = (location.pathname || '').toLowerCase();
			var b = (document.body.className || '').toLowerCase();
			return u.indexOf('vitamin') !== -1 || u.indexOf('papild') !== -1 ||
			       b.indexOf('vitamin') !== -1 || b.indexOf('papild') !== -1;
		}
		function isAnkPage() {
			// antkakliu/pavadeliu kategorijos puslapis (URL ar body klase)
			var u = (location.pathname || '').toLowerCase();
			var b = (document.body.className || '').toLowerCase();
			return u.indexOf('antkakl') !== -1 || u.indexOf('pavadel') !== -1 ||
			       b.indexOf('antkakl') !== -1 || b.indexOf('pavadel') !== -1;
		}
		function isDubPage() {
			// dubeneliu kategorijos puslapis (URL ar body klase)
			var u = (location.pathname || '').toLowerCase();
			var b = (document.body.className || '').toLowerCase();
			return u.indexOf('dubenel') !== -1 || b.indexOf('dubenel') !== -1;
		}
		function isGuolPage() {
			// guoliu/transporto/narvu/kelioniu kategorijos puslapis
			var u = (location.pathname || '').toLowerCase();
			var b = (document.body.className || '').toLowerCase();
			var hit = function(x){ return u.indexOf(x)!==-1 || b.indexOf(x)!==-1; };
			return hit('guoliai') || hit('transportavimo-dezes') || hit('kelioniu-iranga') || hit('narvai');
		}
		function nicheState(f) {
			var tax = (f.getAttribute('data-taxonomy') || '').toLowerCase();
			var t = titleText(f).toLowerCase();
			// Monoprotein - visada nisa
			if (tax.indexOf('monoprotein') !== -1 || t.indexOf('monoprotein') !== -1) return 'always';
			// Grudu tipas - nisa (maiste)
			if (tax.indexOf('grudu') !== -1 || t.indexOf('grudu') !== -1) return 'always';
			// Baltymu saltinis - nisa TIK skanestu puslapyje
			if (tax.indexOf('baltymu') !== -1 || t.indexOf('baltymo saltinis') !== -1 || t.indexOf('baltymu saltinis') !== -1) {
				return isTreatPage() ? 'always' : 'closed';
			}
			// Paskirtis - nisa TIK vitaminu/papildu puslapyje
			if (tax.indexOf('paskirtis') !== -1 || t.indexOf('paskirtis') !== -1) {
				return isVitaminPage() ? 'always' : 'closed';
			}
			// Tipas - nisa antkakliu, dubeneliu IR guoliu/transporto puslapyje
			if (tax === 'pa_tipas' || t === 'tipas') {
				return (isAnkPage() || isDubPage() || isGuolPage()) ? 'always' : 'closed';
			}
			// Speciali mityba - desktop (maiste)
			if (tax.indexOf('speciali') !== -1 || t.indexOf('speciali mityba') !== -1) return 'desktop';
			return 'closed';
		}
		function setInitialState(f) {
			if (f.hasAttribute('data-ps-init')) return;
			f.setAttribute('data-ps-init', '1');
			var s = nicheState(f);
			// nisos akcentas (kairinis bruksnys) - 3 virsutiniai
			if (s === 'always' || s === 'desktop') {
				f.classList.add('ps-niche');
			}
			var open = (s === 'always') ? true : (s === 'desktop') ? !isMobile() : false;
			if (!open) f.classList.add('ps-collapsed');
		}
		function bindToggle(f) {
			var title = f.querySelector('.filter-title');
			if (!title || title.hasAttribute('data-ps-bound')) return;
			title.setAttribute('data-ps-bound', '1');
			title.addEventListener('click', function() {
				f.classList.toggle('ps-collapsed');
				if (!f.classList.contains('ps-collapsed')) {
					setTimeout(function(){ orderOne(f); }, 30);
				}
			});
		}
		function process() {
			document.querySelectorAll('.yith-wcan-filters .yith-wcan-filter').forEach(function(f) {
				setInitialState(f);
				bindToggle(f);
			});
			orderAll();
		}

		// ===== v11: MOBILE off-canvas (#shop-sidebar) islaikymas per YITH AJAX =====
		// SAKNIS (patvirtinta gyvai per DevTools): YITH AJAX perpiesia turinio
		// bloka ir SUNAIKINA #shop-sidebar (off-canvas taikini). Flatsome mygtuko
		// (data-open="#shop-sidebar") handleris suveikia, bet Magnific Popup

		// #shop-sidebar i <body>, kad YITH AJAX jo nebepasiektu/nesunaikintu.
