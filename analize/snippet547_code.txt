// Helper: grąžina choice info bet kuriam produktui (tėviniam arba paslėptam)
// Normalizuoja config į v2 grupių struktūrą: {group_key: {label, gramaturos: {gram: {size: {product_id, price}}}}}
// v1 (senas, {gram:{size}}) automatiškai įvyniojamas į vieną numatytą grupę "_default".
// Grąžina: ['parent_id', 'config'(normalized), 'is_multigroup'=>bool, 'init_group', 'init_gram', 'init_size', 'is_hidden']
function petshop_normalize_choice_config($config) {
    if (!is_array($config) || empty($config)) return null;
    // v2 atpažinimas: pirmas elementas turi 'gramaturos' raktą
    $first = reset($config);
    $is_v2 = is_array($first) && isset($first['gramaturos']);
    if ($is_v2) {
        return array('config' => $config, 'is_multigroup' => (count($config) > 1));
    }
    // v1 → įvyniojam į vieną grupę
    $wrapped = array('_default' => array('label' => '', 'gramaturos' => $config));
    return array('config' => $wrapped, 'is_multigroup' => false);
}

function petshop_get_choice_info($pid) {
    if (!$pid) return null;
    // 1. Tėvinis choice?
    if (get_post_meta($pid, '_petshop_is_choice_bundle', true) === 'yes') {
        $raw = json_decode(get_post_meta($pid, '_petshop_choice_config', true), true);
        $norm = petshop_normalize_choice_config($raw);
        if (!$norm) return null;
        return [
            'parent_id'=>$pid, 'config'=>$norm['config'], 'is_multigroup'=>$norm['is_multigroup'],
            'init_group'=>null, 'init_gram'=>null, 'init_size'=>null, 'is_hidden'=>false,
            'no_gram'=>(get_post_meta($pid,'_petshop_choice_no_gram',true)==='yes')
        ];
    }
    // 2. Paslėptas choice? (turi atgalinę nuorodą į tėvinį)
    $parent = get_post_meta($pid, '_petshop_choice_parent', true);
    if ($parent) {
        $raw = json_decode(get_post_meta($parent, '_petshop_choice_config', true), true);
        $norm = petshop_normalize_choice_config($raw);
        if (!$norm) return null;
        return [
            'parent_id' => intval($parent),
            'config' => $norm['config'],
            'is_multigroup' => $norm['is_multigroup'],
            'init_group' => get_post_meta($pid, '_petshop_choice_group', true) ?: '_default',
            'init_gram' => get_post_meta($pid, '_petshop_choice_gram', true),
            'init_size' => get_post_meta($pid, '_petshop_choice_size', true),
            'is_hidden' => true,
            'hidden_id' => $pid,
            'no_gram' => (get_post_meta(intval($parent),'_petshop_choice_no_gram',true)==='yes')
        ];
    }
    return null;
}

// Įjungiam tėviniams IR paslėptiems choice produktams
add_action('woocommerce_before_single_product', function() {
    global $product;
    if (!$product) return;
    $pid = is_object($product) ? $product->get_id() : 0;
    $info = petshop_get_choice_info($pid);
    if (!$info) return;
    $GLOBALS['petshop_is_choice_page'] = $pid;
    $GLOBALS['petshop_choice_info'] = $info;
});

// Body klasė choice puslapiams (fullwidth, kai nėra nuotraukos)
add_filter('body_class', function($classes) {
    if (!is_product()) return $classes;
    global $post;
    if (!$post) return $classes;
    $info = petshop_get_choice_info($post->ID);
    if (!$info) return $classes;
    $classes[] = 'petshop-choice-page';
    $classes[] = 'petshop-choice-fullwidth'; // visada fullwidth (nuotrauka slepiama)
    return $classes;
});

                . '<span class="psc-btn-big">'.esc_html($glabel).'</span>'
                . '</button>';
        }
        echo '</div></div>';

    // === Konservo dydis mygtukai (rodo numatytosios grupės gramatūras; JS atnaujins) ===
    // v17: be gramatūros režime paslepiam sekciją (mygtukas lieka, kad JS nustatytų currentGram)

    echo '<div class="psc-section"'.$gram_style.'><div class="psc-label">Konservo dydis</div><div class="psc-btn-row psc-gram-row">';
    foreach ($gramaturos as $g) {

        echo '<button type="button" class="psc-btn psc-gram-btn'.$active.'" data-gram="'.esc_attr($g).'">'
            . '<span class="psc-btn-big">'.esc_html($g).' g</span>'
            . '</button>';
    }
    echo '</div></div>';

    // === Rinkinio dydis mygtukai ===

        $active = ($s == $default_size) ? ' psc-active' : '';

add_action('wp_footer', function() {
    if (empty($GLOBALS['petshop_is_choice_page'])) return;
    // v18: konfigą imam iš TĖVINIO (paslėpto produkto puslapy config yra ant tėvinio, ne ant paties)
    $info = isset($GLOBALS['petshop_choice_info']) ? $GLOBALS['petshop_choice_info'] : null;
    $config_src = ($info && !empty($info['parent_id'])) ? $info['parent_id'] : $GLOBALS['petshop_is_choice_page'];
    $config_raw = get_post_meta($config_src, '_petshop_choice_config', true);
    ?>
    <script>
    (function(){
        var constructor = document.querySelector('.petshop-choice-constructor');
        if (!constructor) return;
        var rawConfig = <?php echo $config_raw ?: '{}'; ?>;

        // Normalizuojam config į grupių struktūrą (v1 → viena "_default" grupė)
        var config;
        (function(){
            var keys = Object.keys(rawConfig);
            var first = keys.length ? rawConfig[keys[0]] : null;
            var isV2 = first && first.gramaturos;
            if (isV2) { config = rawConfig; }
            else { config = { '_default': { label: '', gramaturos: rawConfig } }; }
        })();

        // Grąžina aktyvios grupės gramatūras
        function groupGram(gk){ return config[gk] ? config[gk].gramaturos : {}; }

        var currentGroup = constructor.querySelector('.psc-group-btn.psc-active')?.dataset.group
            || Object.keys(config)[0];
        var currentGram = constructor.querySelector('.psc-gram-btn.psc-active')?.dataset.gram;
        var currentSize = constructor.querySelector('.psc-size-btn.psc-active')?.dataset.size;

        // v16: dinamiškai perkuriam gram/size mygtukus (skirtingos gramatūros/grupės = skirtingi dydžiai)
        function attachGramClick(btn, g){
            btn.addEventListener('click', function(){
                constructor.querySelectorAll('.psc-gram-btn').forEach(function(b){ b.classList.remove('psc-active'); });
                btn.classList.add('psc-active');
                currentGram = g;
                rebuildSizes();
                updateView();
            });
        }
        function attachSizeClick(btn, sz){
            btn.addEventListener('click', function(){
                constructor.querySelectorAll('.psc-size-btn').forEach(function(b){ b.classList.remove('psc-active'); });
                btn.classList.add('psc-active');
                currentSize = sz;
                updateView();
            });
        }
        function rebuildGrams(){
            var gg = groupGram(currentGroup);
            var grams = Object.keys(gg).sort(function(a,b){return parseInt(a)-parseInt(b);});
            if (grams.indexOf(currentGram) < 0) currentGram = grams[0];
            var row = constructor.querySelector('.psc-gram-row');
            if (!row) return;
            row.innerHTML = '';
            grams.forEach(function(g){
                var btn = document.createElement('button');
                btn.type = 'button';
                btn.className = 'psc-btn psc-gram-btn' + (g === currentGram ? ' psc-active' : '');
                btn.dataset.gram = g;
                btn.innerHTML = '<span class="psc-btn-big">' + g + ' g</span>';
                attachGramClick(btn, g);
                row.appendChild(btn);
            });
        }
        function rebuildSizes(){
            var gg = groupGram(currentGroup);
            var gd = gg[currentGram] || {};
            var sizes = Object.keys(gd).sort(function(a,b){return parseInt(a)-parseInt(b);});
            if (sizes.indexOf(currentSize) < 0) currentSize = sizes[0];
            var row = constructor.querySelector('.psc-size-row');
            if (!row) return;
            row.innerHTML = '';
            sizes.forEach(function(sz){
                var info = gd[sz];
                var perUnit = info ? (info.price / parseInt(sz)).toFixed(2).replace('.', ',') : '';
                var btn = document.createElement('button');
                btn.type = 'button';
                btn.className = 'psc-btn psc-size-btn' + (sz === currentSize ? ' psc-active' : '');
                btn.dataset.size = sz;
                btn.innerHTML = '<span class="psc-btn-big">' + sz + ' vnt.</span><span class="psc-btn-small psc-per-unit">' + perUnit + ' €/vnt.</span>';
                attachSizeClick(btn, sz);
                row.appendChild(btn);
            });
        }

        function updateView() {
            var gg = groupGram(currentGroup);
            // Rodom tik atitinkančią formą (grupė + gram + size)
            constructor.querySelectorAll('.psc-form').forEach(function(f){
                var fg = f.dataset.group || '_default';
                f.style.display = (fg === currentGroup && f.dataset.gram === currentGram && f.dataset.size === currentSize) ? '' : 'none';
            });
            // Atnaujinam kainą
            var info = gg[currentGram] && gg[currentGram][currentSize];
            if (info) {
                var priceEl = constructor.querySelector('.psc-box-price');
                if (priceEl) priceEl.textContent = info.price.toFixed(2).replace('.', ',') + ' €';
                var targetEl = constructor.querySelector('.psc-target-count');
                if (targetEl) targetEl.textContent = currentSize;
            }
            // Atnaujinam dydžio mygtukų €/vnt pagal dabartinę grupę+gramatūrą
            constructor.querySelectorAll('.psc-size-btn').forEach(function(btn){
                var sz = btn.dataset.size;
                var inf = gg[currentGram] && gg[currentGram][sz];
                if (inf) {
                    var perUnit = (inf.price / parseInt(sz)).toFixed(2).replace('.', ',');
                    var puEl = btn.querySelector('.psc-per-unit');
                    if (puEl) puEl.textContent = perUnit + ' €/vnt.';
                }
            });
            moveButton();
            hideOutOfStock();
            updateSummary();

        function visibleForm() {

        // Tikro MnM mygtuko NEKELIAM iš formos (kitaip submit neveikia — mygtukas turi būti <form> viduje).

        function moveButton() {
            if (!ctaSlot) return;
            var form = visibleForm();
            if (!form) return;
            // Sukuriam proxy mygtuką (jei dar nėra)
            var proxy = ctaSlot.querySelector('.psc-proxy-cta');
            if (!proxy) {
                proxy = document.createElement('button');
                proxy.type = 'button';
                proxy.className = 'psc-proxy-cta';
                proxy.textContent = 'Į krepšelį';
                ctaSlot.appendChild(proxy);
                proxy.addEventListener('click', function(){
                    if (proxy.classList.contains('disabled')) return;

                    var realBtn = vf ? vf.querySelector('.single_add_to_cart_button') : null;
                    if (realBtn) realBtn.click();

        var sumSelEl = constructor.querySelector('.psc-sum-selected');
        var sumLeftEl = constructor.querySelector('.psc-sum-left');
        var progressFill = constructor.querySelector('.psc-progress-fill');
        function ltPlural(n){
            n = Math.abs(n);
            if (n === 1 || (n % 10 === 1 && n % 100 !== 11)) return 'konservą';
            var d = n % 10, dd = n % 100;
            if (d >= 2 && d <= 9 && (dd < 10 || dd > 20)) return 'konservus';
            return 'konservų';
        }
        function updateSummary() {
            var form = visibleForm();
            if (!form) return;
            var inputs = form.querySelectorAll('.mnm_child_products input.qty, .mnm_child_products input[type=number]');
            var sel = 0;
            inputs.forEach(function(inp){ var v = parseInt(inp.value); if (!isNaN(v)) sel += v; });
            var need = parseInt(currentSize) || 0;
            if (sumSelEl) sumSelEl.textContent = sel + ' / ' + need;
            var left = Math.max(0, need - sel);
            if (progressFill) progressFill.style.width = (need > 0 ? Math.min(100, Math.round(sel/need*100)) : 0) + '%';
            if (sumLeftEl) {
                if (sel >= need && need > 0) { sumLeftEl.textContent = 'Rinkinys pilnas'; sumLeftEl.style.color = '#0F6E56'; }
                else { sumLeftEl.textContent = left; sumLeftEl.style.color = ''; }
            }
            // Proxy mygtuko būsena — MnM pats nevaldo, valdom mes pagal progresą
            var btn = ctaSlot ? ctaSlot.querySelector('.psc-proxy-cta') : null;
            if (btn) {
                if (sel >= need && need > 0) {
                    btn.classList.remove('disabled');
                    btn.textContent = 'Į krepšelį';
                } else {
                    btn.classList.add('disabled');
                    btn.textContent = 'Dar pasirinkite ' + left + ' ' + ltPlural(left);
                }
            }
        }

                rebuildGrams();
                rebuildSizes();
                updateView();

        if (window.jQuery) {

                setTimeout(function(){ moveButton(); updateSummary(); }, 50);
            });
            jQuery(document).on('change keyup click', '.psc-form input.qty, .psc-form .mnm_child_products button, .psc-form .plus, .psc-form .minus', function(){
                setTimeout(updateSummary, 80);
            });
        }

        // ETAPAS 6: slepia out-of-stock konservus (eilutes su "Neturime"/disabled input)
        function hideOutOfStock() {
            var form = visibleForm();
            if (!form) return;
            var rows = form.querySelectorAll('.mnm_child_products tr, .mnm_child_products .mnm_item');
            rows.forEach(function(row){
                if (row.querySelector('th')) return;
                var input = row.querySelector('input.qty, input[type=number]');
                var txt = row.textContent || '';
                var isOOS = (input && (input.disabled || input.max === '0')) ||
                            (/Neturime|Out of stock|Šiuo metu nėra/i.test(txt) && !(input && !input.disabled));
                row.style.display = isOOS ? 'none' : '';
            });
        }

        rebuildGrams();
        rebuildSizes();
        updateView();
        // MnM inicijuojasi vėlai — kartotinai
        window.addEventListener('load', function(){ setTimeout(function(){ moveButton(); hideOutOfStock(); updateSummary(); }, 400); });
        var _t=0, _iv=setInterval(function(){ moveButton(); hideOutOfStock(); updateSummary(); if(++_t>=8) clearInterval(_iv); }, 400);
    })();
