(function () {
	window.dataLayer = window.dataLayer || [];
	function gtag() { window.dataLayer.push(arguments); }

	/* ---- 1. CONSENT MODE v2 DEFAULT (privalo buti pirmas) ---- */
	gtag('consent', 'default', {
		'ad_storage': 'denied',
		'ad_user_data': 'denied',
		'ad_personalization': 'denied',
		'analytics_storage': 'denied',
		'functionality_storage': 'granted',
		'personalization_storage': 'granted',
		'security_storage': 'granted',
		'wait_for_update': 500
	});
	gtag('set', 'ads_data_redaction', true);
	gtag('set', 'url_passthrough', true);

	/* ---- 2. Pagalbines ---- */
	function cookieVal(name) {
		var m = document.cookie.match(new RegExp('(?:^|;\\s*)' + name.replace(/[-]/g, '\\-') + '=([^;]*)'));
		return m ? decodeURIComponent(m[1]) : null;
	}

	function hasConsent(cat) {
		if (typeof window.cmplz_has_consent === 'function') {
			try { return !!window.cmplz_has_consent(cat); } catch (e) {}
		}
		return cookieVal('cmplz_' + cat) === 'allow';
	}

	function sendConsentUpdate(source) {
		var stats = hasConsent('statistics');
		var mkt   = hasConsent('marketing');

		gtag('consent', 'update', {
			'analytics_storage':  stats ? 'granted' : 'denied',
			'ad_storage':         mkt   ? 'granted' : 'denied',
			'ad_user_data':       mkt   ? 'granted' : 'denied',
			'ad_personalization': mkt   ? 'granted' : 'denied'
		});

		window.dataLayer.push({
			event: 'cmplz_consent_update',
			cmplz_statistics: stats ? 'granted' : 'denied',
			cmplz_marketing:  mkt   ? 'granted' : 'denied',
			cmplz_source: source || 'unknown'
		});
	}

	/* ---- 3. Pakartotinis lankytojas: sutikimas jau issaugotas ---- */
	var hasConsentSnapshot = hasConsent('statistics');
	if (cookieVal('cmplz_banner-status') !== null) {
		sendConsentUpdate('cookie');
	}

	/* ---- 4. Naujas sutikimas arba pakeitimas ---- */
	var timer = null;
	function schedule() {
		if (timer) { clearTimeout(timer); }
		timer = setTimeout(function () {
			var wasGranted = hasConsentSnapshot;
			sendConsentUpdate('event');
			var nowGranted = hasConsent('statistics');

			/* Replay TIK jei sutikimas ka tik pasikeite denied -> granted.
			 * Kitaip cmplz_fire_categories (fire'ina kas puslapi) dubliuotu ivykius. */
			if (!wasGranted && nowGranted) { replayEcommerce(); }
			hasConsentSnapshot = nowGranted;
		}, 60);
	}
	document.addEventListener('cmplz_status_change', schedule);
	document.addEventListener('cmplz_fire_categories', schedule);

	/* ---- 5. Ecommerce replay ----
	 * view_item / view_cart ivyksta puslapio uzkrovime, pries sutikima.
	 * Po sutikimo GA4 tag'as jau nebefire'ins, nes trigger'is praejo.
	 * Todel paskutini ecommerce event'a pakartojam VIENA karta. */
	var replayed = false;
	function replayEcommerce() {
		if (replayed) { return; }
		var dl = window.dataLayer || [];
		var last = null;
		for (var i = dl.length - 1; i >= 0; i--) {
			var x = dl[i];
			if (x && x.event && x.ecommerce && x.event !== 'cmplz_consent_update') { last = x; break; }
		}
		if (!last) { return; }
		replayed = true;
		var copy = JSON.parse(JSON.stringify(last));
		copy.cmplz_replay = true;
		window.dataLayer.push({ ecommerce: null });
		window.dataLayer.push(copy);
	}
})();

			return hit('guoliai') || hit('transportavimo-dezes') || hit('kelioniu-iranga') || hit('narvai');
		}
		function nicheState(f) {
			var tax = (f.getAttribute('data-taxonomy') || '').toLowerCase();
			var t = titleText(f).toLowerCase();
			// Monoprotein - visada nisa
			if (tax.indexOf('monoprotein') !== -1 || t.indexOf('monoprotein') !== -1) return 'always';
			// Grudu tipas - nisa (maiste)
			if (tax.indexOf('grudu') !== -1 || t.indexOf('grudu') !== -1) return 'always';
			// Baltymu saltinis - nisa TIK skanestu puslapyje
			if (tax.indexOf('baltymu') !== -1 || t.indexOf('baltymo saltinis') !== -1 || t.indexOf('baltymu saltinis') !== -1) {
				return isTreatPage() ? 'always' : 'closed';
			}
			// Paskirtis - nisa TIK vitaminu/papildu puslapyje
			if (tax.indexOf('paskirtis') !== -1 || t.indexOf('paskirtis') !== -1) {
				return isVitaminPage() ? 'always' : 'closed';
			}
			// Tipas (pa_tipas, pa_zaislo_tipas, bet koks pa_*_tipas) - nisa VISADA kur rodomas.
			// Preset si "tipo" filtra rodo tik aksesuaru kategorijose (ne maiste),
			// todel taksonomijos-agnostiskas 'always' saugus -> PILNAS uzsaldomas.
			if (tax === 'pa_tipas' || /_tipas$/.test(tax) || t === 'tipas' || t.indexOf('tipas') !== -1) {
				return 'always';
			}
			// Speciali mityba - desktop (maiste)
			if (tax.indexOf('speciali') !== -1 || t.indexOf('speciali mityba') !== -1) return 'desktop';
			return 'closed';
		}
		function setInitialState(f) {
			if (f.hasAttribute('data-ps-init')) return;
			f.setAttribute('data-ps-init', '1');
			var s = nicheState(f);
			// nisos akcentas (kairinis bruksnys) - 3 virsutiniai
			if (s === 'always' || s === 'desktop') {
				f.classList.add('ps-niche');
			}
			var open = (s === 'always') ? true : (s === 'desktop') ? !isMobile() : false;
			if (!open) f.classList.add('ps-collapsed');
		}
		function bindToggle(f) {
			var title = f.querySelector('.filter-title');
			if (!title || title.hasAttribute('data-ps-bound')) return;
			title.setAttribute('data-ps-bound', '1');
			title.addEventListener('click', function() {
				f.classList.toggle('ps-collapsed');
				if (!f.classList.contains('ps-collapsed')) {
					setTimeout(function(){ orderOne(f); }, 30);
				}
			});
		}
		function process() {
			document.querySelectorAll('.yith-wcan-filters .yith-wcan-filter').forEach(function(f) {
				setInitialState(f);
				bindToggle(f);
			});
			orderAll();
		}

		// ===== v11: MOBILE off-canvas (#shop-sidebar) islaikymas per YITH AJAX =====
		// SAKNIS (patvirtinta gyvai per DevTools): YITH AJAX perpiesia turinio
		// bloka ir SUNAIKINA #shop-sidebar (off-canvas taikini). Flatsome mygtuko
		// (data-open="#shop-sidebar") handleris suveikia, bet Magnific Popup
		// nebeturi i ka nukreipti -> tyliai neatsidaro. Sprendimas: perkelti
		// #shop-sidebar i <body>, kad YITH AJAX jo nebepasiektu/nesunaikintu.
		// YITH atnaujina filtru widget'a JO VIDUJE (busena lieka teisinga).
		function persistOffcanvas() {
			// v12: TIK off-canvas plociuose (< 850px). Desktop'e #shop-sidebar
			// privalo likti savo kolonoje - kitaip dingsta desktop filtras.
			if (window.innerWidth >= 850) return;
			var oc = document.getElementById('shop-sidebar');
			if (!oc) return;
			// jei siuo metu atidarytas (Magnific perkele i .mfp-content) - neliesti
			if (oc.closest && oc.closest('.mfp-content')) return;
			if (oc.parentElement !== document.body) {
				document.body.appendChild(oc);
			}
		}
		// Po YITH AJAX: uztikrinam taikini + perstilizuojam widget'a (innerHTML pasikeite)
		function afterAjax() {
			setTimeout(function(){
				persistOffcanvas();
				process();
			}, 30);
		}

		function init() {
			process();
			persistOffcanvas();
			var container = document.querySelector('.filters-container, aside, .widget-area') || document.body;
			if (container && window.MutationObserver) {
				var timer = null;
				var obs = new MutationObserver(function() {
					clearTimeout(timer);
					timer = setTimeout(process, 120);
				});
				obs.observe(container, { childList: true, subtree: true });
			}

			// v11: po YITH AJAX - islaikom #shop-sidebar taikini + perstilizuojam
			if (window.jQuery) {
				window.jQuery(document).on(
					'yith-wcan-ajax-loaded yith_wcan_ajax_loaded yith-wcan-ajax-success ajaxComplete',
					afterAjax
				);
			}

			document.addEventListener('click', function(e) {
				var t = e.target; if (!t) return;
				var txt = (t.textContent || '').toLowerCase();
				var cls = (t.className || '').toString().toLowerCase();
				if (txt.indexOf('show more') !== -1 || txt.indexOf('show less') !== -1 ||
				    cls.indexOf('show-more') !== -1 || cls.indexOf('shorten') !== -1) {
					setTimeout(orderAll, 60);
					setTimeout(orderAll, 250);
				}
			}, true);
			var ticks = 0;
			var iv = setInterval(function() {
				orderAll();
				persistOffcanvas();
				if (++ticks >= 8) clearInterval(iv);
			}, 500);
		}
		if (document.readyState === 'loading') {
			document.addEventListener('DOMContentLoaded', init);
		} else { init(); }
	})();
